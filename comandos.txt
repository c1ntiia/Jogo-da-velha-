git --version
    ver a versão instalada do git bash

git config user.name
    configurar/adicionar um nome de usuário

git config user.email
    configurar/adicionar um email

git init
    inicair o repositorio

git branch -M
    mudar nome da branch

git branch -a
    listar branchs existentes

git branch 
    criar nova branch

git status
    mostra informações do repositorio (branch, commits)

git add .
    adicionar os arquivos locais ao repositorio

git commit -m
    comentar a mudança que foi feita no repositorio

git log
    histórico dos commits

git checkout -b
    criar nova branch e ir pra branch criada

git switch/checkout
    mudar pra branch

git clone <url>
    clonar repositorio existente

git push -u origin main
    primeira vez subindo repositorio remoto

git pull
    trazer alterações do repositorio remoto para o local

git push
    subir alterações feitas para o repositorio remoto

git merge
    mescar as alterações feitas em uma branch a outra
